    <button class="btn" onclick="switchLevel(0)">Dữ liệu gốc</button>
    <button class="btn" onclick="switchLevel(1)">Cấp độ 1</button>
    <button class="btn" onclick="switchLevel(2)">Cấp độ 2</button>
    <button class="btn" onclick="switchLevel(3)">Cấp độ 3</button>

<script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>
<script src="https://cdnjs.cloudflare.com/ajax/libs/leaflet.draw/1.0.4/leaflet.draw.js"></script>

<script>
    // 1. Khởi tạo bản đồ tại khu vực Phú Hòa, Tuy Hòa (Dựa trên hình ảnh)
    var map = L.map('map').setView([13.100, 109.250], 13);

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
        attribution: '© OpenStreetMap contributors'
    }).addTo(map);

    // 2. Tọa độ khớp ảnh (Bounding Box) - Bạn cần tinh chỉnh tọa độ này theo thực tế
    // Dựa vào kinh độ/vĩ độ trên ảnh của bạn: 109°12'E đến 109°18'E và 13°3'N đến 13°9'N
    var imageBounds = [[13.050, 109.200], [13.150, 109.300]];

    // 3. Khai báo các lớp ảnh (Image Overlays)
    // Lưu ý: Đặt các file ảnh CAP1.png, CAP2.png... cùng thư mục với file index.html
    var layers = {
        0: L.imageOverlay('GTHONG.png', imageBounds),
        1: L.imageOverlay('CAP1.png', imageBounds),
        2: L.imageOverlay('CAP2.png', imageBounds),
        3: L.imageOverlay('CAP3.png', imageBounds),
        'dem': L.imageOverlay('MOHINHSODOCAO.png', imageBounds, {opacity: 0.5})
    };

    var currentLayer = null;

    function switchLevel(level) {
        // Xóa lớp hiện tại
        if (currentLayer) map.removeLayer(currentLayer);
        if (map.hasLayer(layers['dem'])) map.removeLayer(layers['dem']);

        // Thêm lớp mới
        currentLayer = layers[level];
        currentLayer.addTo(map);
        
        // Cấp 3 thì hiện thêm bản đồ địa hình DEM để tìm chỗ cao
        if (level === 3) layers['dem'].addTo(map);

        updateTasks(level);
    }

    // 4. Hiển thị nhiệm vụ tương tác (Tương ứng với file docx của bạn)
    function updateTasks(level) {
        var info = document.getElementById('info-box');
        info.style.display = 'block';
        var tasks = "";
        
        if (level === 1) {
            tasks = "<b>Nhiệm vụ Cấp 1:</b><br/>- Xác định vùng nguy cơ [cite: 7]<br/>- Chọn 1 điểm an toàn [cite: 8]<br/>- Tìm lộ trình ngắn nhất [cite: 11]";
        } else if (level === 2) {
            tasks = "<b>Nhiệm vụ Cấp 2:</b><br/>- Kiểm tra lộ trình cũ [cite: 16]<br/>- Vẽ tuyến đường thay thế an toàn [cite: 18]";
        } else if (level === 3) {
            tasks = "<b>Nhiệm vụ Cấp 3:</b><br/>- Tìm vị trí cao trong trường [cite: 23]<br/>- Soạn tin nhắn cầu cứu [cite: 27]";
        }
        info.innerHTML = tasks;
    }

    // 5. Thêm công cụ vẽ (Drawing Tools) để học sinh tương tác
    var drawnItems = new L.FeatureGroup();
    map.addLayer(drawnItems);
    var drawControl = new L.Control.Draw({
        edit: { featureGroup: drawnItems },
        draw: {
            polyline: { shapeOptions: { color: 'red' } }, // Vẽ lộ trình di chuyển
            marker: true, // Đánh dấu điểm an toàn
            polygon: false, circle: false, rectangle: false
        }
    });
    map.addControl(drawControl);

    map.on(L.Draw.Event.CREATED, function (event) {
        var layer = event.layer;
        drawnItems.addLayer(layer);
    });

    // Mặc định chạy Cấp 1
    switchLevel(1);

</script>
